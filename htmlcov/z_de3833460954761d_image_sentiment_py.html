17 def __init__(self): 

18 """Initialize the sentiment detection models."""

25 MODEL_CONFIG["image_captioning"]["model_name"], 

26 cache_dir=self.cache_dir 

27 ) 

28 self.caption_model = BlipForConditionalGeneration.from_pretrained( 

29 MODEL_CONFIG["image_captioning"]["model_name"],

33 # Initialize sentiment classifier 

34 print("Loading sentiment analysis model...") 

35 self.sentiment_analyzer = pipeline( 

36 "sentiment-analysis",

55 def generate_caption(self, image: Image.Image) -> str: 

56 """ 

57 Generate descriptive caption for image using BLIP. 

58  

59 Args: 

60 image: PIL Image object 

61  

62 Returns: 

63 Generated caption string 

64 """ 

65 try: 

66 # Preprocess image 

67 inputs = self.caption_processor(image, return_tensors="pt").to(self.device) 

68 

69 # Generate caption 

70 with torch.no_grad(): 

71 output = self.caption_model.generate(**inputs, max_length=50) 

72 

73 # Decode caption 

74 caption = self.caption_processor.decode(output[0], skip_special_tokens=True) 

75 

76 return caption 

77 

78 except Exception as e: 

79 print(f"Error generating caption: {str(e)}") 

80 return "a scene"

82 def analyze_sentiment_from_text(self, text: str) -> Dict[str, float]: 

83 """ 

84 Analyze sentiment from text description. 

85  

86 Args: 

87 text: Text to analyze 

88  

89 Returns: 

90 Dictionary of sentiment scores 

91 """ 

92 sentiment_scores = {} 

93 

94 # Get base sentiment (positive/negative) 

95 try: 

96 base_sentiment = self.sentiment_analyzer(text)[0] 

97 base_score = base_sentiment['score'] if base_sentiment['label'] == 'POSITIVE' else 1 - base_sentiment['score'] 

98 except: 

99 base_score = 0.5 

100 

101 # Analyze keywords for specific sentiments 

102 text_lower = text.lower() 

103 

104 for sentiment, keywords in self.sentiment_keywords.items(): 

105 score = 0.0 

106 matches = 0 

107 

108 for keyword in keywords: 

109 if keyword in text_lower: 

110 matches += 1 

111 score += 0.2 # Each keyword match adds to score 

112 

113 # Normalize score 

114 if matches > 0: 

115 score = min(score, 1.0) 

116 # Boost with base sentiment 

117 score = (score + base_score) / 2 

118 else: 

119 score = base_score * 0.3 # Low score if no keywords match 

120 

121 sentiment_scores[sentiment] = round(score, 3) 

122 

123 return sentiment_scores

125 def detect_category(self, caption: str) -> str: 

126 """ 

127 Detect image category (food, travel, nature, lifestyle). 

128  

129 Args: 

130 caption: Image caption 

131  

132 Returns: 

133 Category string 

134 """ 

135 caption_lower = caption.lower() 

136 

137 # Category keywords 

138 food_keywords = ["food", "meal", "dish", "plate", "eating", "restaurant", "kitchen", 

139 "cooking", "dessert", "drink", "coffee", "cake", "pizza", "burger"] 

140 travel_keywords = ["travel", "city", "building", "street", "architecture", "landmark", 

141 "hotel", "airport", "beach", "vacation", "trip"] 

142 nature_keywords = ["nature", "tree", "forest", "mountain", "sky", "water", "ocean", 

143 "lake", "river", "flower", "plant", "landscape", "sunset", "sunrise"] 

144 

145 # Count matches 

146 food_score = sum(1 for kw in food_keywords if kw in caption_lower) 

147 travel_score = sum(1 for kw in travel_keywords if kw in caption_lower) 

148 nature_score = sum(1 for kw in nature_keywords if kw in caption_lower) 

149 

150 # Determine category 

151 scores = { 

152 "food": food_score, 

153 "travel": travel_score, 

154 "nature": nature_score, 

155 "lifestyle": 1 # Default baseline 

156 } 

157 

158 return max(scores, key=scores.get)

160 def detect_sentiment(self, image: Image.Image) -> Dict: 

161 """ 

162 Main method to detect sentiment from image. 

163  

164 Args: 

165 image: PIL Image object 

166  

167 Returns: 

168 Dictionary with sentiment, confidence, caption, and category 

169 """ 

170 try: 

171 # Generate caption 

172 caption = self.generate_caption(image) 

173 

174 # Analyze sentiment 

175 sentiment_scores = self.analyze_sentiment_from_text(caption) 

176 

177 # Get top sentiment 

178 top_sentiment = max(sentiment_scores, key=sentiment_scores.get) 

179 confidence = sentiment_scores[top_sentiment] 

180 

181 # Detect category 

182 category = self.detect_category(caption) 

183 

184 return { 

185 "sentiment": top_sentiment, 

186 "confidence": confidence, 

187 "all_sentiments": sentiment_scores, 

188 "caption": caption, 

189 "category": category, 

190 "success": True 

191 } 

192 

193 except Exception as e: 

194 print(f"Error in sentiment detection: {str(e)}") 

195 return { 

196 "sentiment": "happy", 

197 "confidence": 0.5, 

198 "all_sentiments": {}, 

199 "caption": "Unable to analyze image", 

200 "category": "lifestyle", 

201 "success": False, 

202 "error": str(e) 

203 }
